// Función para encontrar la ruta óptima usando el algoritmo A*
const findPath = (start, end, mapData, userPreferences = {}) => {
  const { width, height, obstacles } = mapData;
 
  // Crear la matriz del mapa
  const grid = Array.from({ length: height }, () =>
    Array(width).fill(0)
  );
 
  // Marcar obstáculos en el mapa
  obstacles.forEach(obs => {
    grid[obs.y][obs.x] = 1;
  });
 
  // Marcar áreas a evitar en el mapa según las preferencias del usuario
  if (userPreferences.avoidAreas) {
    userPreferences.avoidAreas.forEach(area => {
      grid[area.y][area.x] = 1;
    });
  }
 
  // Función para calcular la heurística (distancia Manhattan)
  const heuristic = (a, b) => Math.abs(a.x - b.x) + Math.abs(a.y - b.y);
 
  // Clase para representar un nodo
  class Node {
    constructor(position, parent = null) {
      this.position = position;
      this.parent = parent;
      this.g = 0;
      this.h = 0;
      this.f = 0;
    }
  }
 
  const openList = [];
  const closedList = [];
  
  // Nodo inicial
  const startNode = new Node(start);
  const endNode = new Node(end);
 
  openList.push(startNode);
 
  // Encontrar nodos vecinos válidos
  const getNeighbors = (node) => {
    const { x, y } = node.position;
    const neighbors = [];
 
    const directions = [
      { x: 1, y: 0 }, { x: -1, y: 0 },
      { x: 0, y: 1 }, { x: 0, y: -1 }
    ];
 
    directions.forEach(dir => {
      const newX = x + dir.x;
      const newY = y + dir.y;
      if (
        newX >= 0 && newX < width &&
        newY >= 0 && newY < height &&
        grid[newY][newX] !== 1
      ) {
        neighbors.push(new Node({ x: newX, y: newY }, node));
      }
    });
 
    return neighbors;
  };
 
  while (openList.length > 0) {
    // Obtener el nodo con el menor valor f
    const currentNode = openList.reduce((prev, curr) => 
      prev.f < curr.f ? prev : curr
    );
 
    // Mover el nodo actual de la lista abierta a la lista cerrada
    openList.splice(openList.indexOf(currentNode), 1);
    closedList.push(currentNode);
 
    // Si llegamos al nodo final, reconstruir el camino
    if (currentNode.position.x === endNode.position.x && currentNode.position.y === endNode.position.y) {
      const path = [];
      let current = currentNode;
      while (current) {
        path.push(current.position);
        current = current.parent;
      }
      return path.reverse();
    }
 
    // Obtener los vecinos del nodo actual
    const neighbors = getNeighbors(currentNode);
 
    neighbors.forEach(neighbor => {
      if (closedList.find(closedNode => closedNode.position.x === neighbor.position.x && closedNode.position.y === neighbor.position.y)) {
        return;
      }
 
      neighbor.g = currentNode.g + 1;
      neighbor.h = heuristic(neighbor.position, endNode.position);
      neighbor.f = neighbor.g + neighbor.h;
 
      if (openList.find(openNode => openNode.position.x === neighbor.position.x && openNode.position.y === neighbor.position.y && neighbor.g > openNode.g)) {
        return;
      }
 
      openList.push(neighbor);
    });
  }
 
  return []; // Retornar una lista vacía si no se encuentra un camino
};
 
// Función para calcular la distancia total de la ruta
const calculateDistance = (path) => {
  let distance = 0;
  for (let i = 0; i < path.length - 1; i++) {
    distance += Math.sqrt(
      Math.pow(path[i + 1].x - path[i].x, 2) +
      Math.pow(path[i + 1].y - path[i].y, 2)
    );
  }
  return distance;
};
 
// Función para calcular la duración estimada de la ruta
const calculateDuration = (distance, speed = 1) => {
  return distance / speed; // Asumiendo velocidad constante
};
 
export { findPath, calculateDistance, calculateDuration };
